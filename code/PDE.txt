# Bistable 2D System in XPPAUT

# Set up parameters for the polar dynamics
param mu = 0.5
param a = 1.0
param b = 0.5
param c = 0.1
param q = 0.2
param d = 0.05
param e = 0.05

# Define the time derivative equations
r' = r * (mu - a * r^2 + b * r^4 - c * r^6)
theta' = 1 + q * r^2

# Set initial conditions
r(0) = 0.5
theta(0) = 0

# Define plot settings
@ total=1000, dt=0.05
@ xp=r*cos(theta), yp=r*sin(theta)
@ xlo=-2, xhi=2, ylo=-2, yhi=2

done
