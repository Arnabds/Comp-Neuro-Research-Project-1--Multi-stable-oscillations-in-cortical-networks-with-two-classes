# Bistable 2D System in XPPAUT

# Set up parameters for the polar dynamics
param mu = 0.5
param a = 1.2
param b = 0.7
param c = 0.1
param q = -0.1


lam(r) = mu - a*r +b*r^2 -c*r^3  
omeg(r) = 1+q*r
f(u,v) = lam(u^2+v^2)*u - omeg(u^2+v^2)*v
g(u,v) = lam(u^2+v^2)*v + omeg(u^2+v^2)*u


u' = f(u,v) 
v' = g(u,v)


# Define plot settings
@ xp = u, yp = v, xlo=-3, xhi=3, ylo=-3, yhi=3
